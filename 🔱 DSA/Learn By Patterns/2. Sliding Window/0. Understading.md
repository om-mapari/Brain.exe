Fixed size window template

```cpp
int fixedSlidingWindow(vector<int>& arr, int k) {
    int n = arr.size();
    if (k > n) return -1;

    int windowSum = 0;

    // Step 1: Build first window
    for (int i = 0; i < k; i++) {
        windowSum += arr[i];
    }

    int maxResult = windowSum;

    // Step 2: Slide window
    for (int i = k; i < n; i++) {
        windowSum += arr[i];       // add new element to window
        windowSum -= arr[i - k];   // remove old element

        maxResult = max(maxResult, windowSum);
    }

    return maxResult;
}
```
